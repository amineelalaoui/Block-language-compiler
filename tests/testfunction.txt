 pgcd {
  const int test = 0;
   int pgcd(int a , int b ) {
     while (a * b != test) {
       if ( a > b ) {
 	int na = a-b;
 	a = na;
       } else {
 	int nb = b-a;
 	b = nb;
       }
     }
     int res = a;
     if (res == test) {
       res = b;
     }
     return res;
   }

    int a = 47;
    int b = 53;
   print pgcd(a,b);
 }
