pgcd {
    <int,int> c = {47,53};
    const int test = 0;
    int x = fst c;
    int y = snd c;
    while (x * y != test) {
        if ( x > y ) {
            int na = x-y;
            x = na;
        } else {
            int nb = y-x;
            y = nb;
        }
    }
    int res = x;
    if (res == test) {
        res = y;
    }
    print res;
}
